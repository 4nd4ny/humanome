// Fonction de correction d'encodage UTF-8
const fixUTF8Encoding = (text) => {
  if (!text) return text;
  const corrections = {
    'Ã©': 'é', 'Ã¨': 'è', 'Ã ': 'à', 'Ã´': 'ô', 'Ã¢': 'â', 'Ãª': 'ê',
    'Ã®': 'î', 'Ã§': 'ç', 'Ã¹': 'ù', 'Ã»': 'û', 'Ã¯': 'ï', 'Ã«': 'ë',
    'Ã€': 'À', 'Ã‰': 'É', 'Ãˆ': 'È', 'Ã"': 'Ó', 'Ã‚': 'Â', 'ÃŠ': 'Ê',
    'ÃŽ': 'Î', 'Ã‡': 'Ç', 'Ã™': 'Ù', 'Ã›': 'Û', 'Ã': 'Ï', 'Ã‹': 'Ë',
    'Å"': 'œ', 'Ã¯': 'ï'
  };
  
  let correctedText = text;
  for (const [bad, good] of Object.entries(corrections)) {
    correctedText = correctedText.replace(new RegExp(bad, 'g'), good);
  }
  return correctedText;
};

// Parser XML
const parseXMLData = (xmlString) => {
  const correctedXML = fixUTF8Encoding(xmlString);
  const parser = new DOMParser();
  const xmlDoc = parser.parseFromString(correctedXML, "text/xml");
  
  const cartoElement = xmlDoc.querySelector('CARTO');
  const learner = fixUTF8Encoding(cartoElement?.getAttribute('LEARNER') || '');
  const examId = fixUTF8Encoding(cartoElement?.getAttribute('EXAMID') || '');
  const date = fixUTF8Encoding(cartoElement?.getAttribute('DATE') || '');
  const teacher = fixUTF8Encoding(cartoElement?.getAttribute('TEACHER') || '');

  // Parse metadata
  const metadataElement = xmlDoc.querySelector('METADATA');
  const metadata = {
    totalPoints: parseFloat(metadataElement?.querySelector('TOTAL_POINTS')?.textContent || 0),
    heuresFormation: parseFloat(metadataElement?.querySelector('HEURES_FORMATION')?.textContent || 0),
    activitesRealisees: parseInt(metadataElement?.querySelector('ACTIVITES_REALISEES')?.textContent || 0),
    niveauMoyen: parseFloat(metadataElement?.querySelector('NIVEAU_MOYEN')?.textContent || 0),
    domaineFort: fixUTF8Encoding(metadataElement?.querySelector('DOMAINE_FORT')?.textContent || ''),
    progressionTrimestre: metadataElement?.querySelector('PROGRESSION_TRIMESTRE')?.textContent || ''
  };

  // Collecter toutes les traces
  const allTraces = [];
  
  // Parser récursivement les groupes
  const parseGroup = (groupElement, parent = null, depth = 0) => {
    const id = fixUTF8Encoding(groupElement.getAttribute('ID'));
    const color = groupElement.getAttribute('COLOR');
    
    const node = {
      id,
      parent,
      depth,
      points: 0,
      color: color || (parent?.color),
      children: [],
      isLeaf: false,
      niveau: 0,
      competenceCount: 0
    };

    // Parse COMPETENCE elements
    const competenceElements = Array.from(groupElement.children).filter(child => child.tagName === 'COMPETENCE');
    competenceElements.forEach(compEl => {
      const competenceId = fixUTF8Encoding(compEl.getAttribute('ID'));
      const competenceNode = {
        id: competenceId,
        parent: node,
        depth: depth + 1,
        points: parseFloat(compEl.getAttribute('POINTS') || 0),
        niveau: parseInt(compEl.getAttribute('NIVEAU') || 1),
        description: fixUTF8Encoding(compEl.getAttribute('DESCRIPTION')),
        color: node.color,
        children: [],
        isLeaf: true,
        traces: []
      };
      
      // Parser les traces de cette compétence
      const traceElements = Array.from(compEl.children).filter(child => child.tagName === 'TRACE');
      traceElements.forEach(traceEl => {
        const trace = {
          id: fixUTF8Encoding(traceEl.getAttribute('ID')),
          date: traceEl.getAttribute('DATE'),
          points: parseFloat(traceEl.getAttribute('POINTS') || 0),
          description: fixUTF8Encoding(traceEl.getAttribute('DESCRIPTION')),
          type: fixUTF8Encoding(traceEl.getAttribute('TYPE')),
          url: traceEl.getAttribute('URL'),
          competence: competenceId,
          domain: node.id,
          domainColor: node.color
        };
        competenceNode.traces.push(trace);
        allTraces.push(trace);
      });
      
      node.children.push(competenceNode);
      node.points += competenceNode.points;
      node.niveau = Math.max(node.niveau, competenceNode.niveau);
      node.competenceCount++;
    });

    // Parse nested GROUP elements
    const groupElements = Array.from(groupElement.children).filter(child => child.tagName === 'GROUP');
    groupElements.forEach(groupEl => {
      const childNode = parseGroup(groupEl, node, depth + 1);
      node.children.push(childNode);
      node.points += childNode.points;
      node.competenceCount += childNode.competenceCount;
      if (childNode.niveau > node.niveau) {
        node.niveau = childNode.niveau;
      }
    });

    return node;
  };

  const rootNode = {
    id: learner,
    parent: null,
    depth: 0,
    points: 0,
    children: [],
    isLeaf: false,
    niveau: 0,
    competenceCount: 0
  };

  const topGroups = xmlDoc.querySelectorAll('CARTO > GROUP');
  topGroups.forEach(groupEl => {
    const childNode = parseGroup(groupEl, rootNode, 1);
    rootNode.children.push(childNode);
    rootNode.points += childNode.points;
    rootNode.competenceCount += childNode.competenceCount;
  });

  // Trier les traces par date
  allTraces.sort((a, b) => new Date(a.date) - new Date(b.date));

  return { root: rootNode, learner, examId, date, teacher, metadata, traces: allTraces };
};

// Composant Secteur Circulaire
const CircularSector = ({ 
  sector, centerX, centerY, innerRadius, outerRadius, 
  startAngle, endAngle, color, isSelected, isHovered, isActive,
  onClick, onMouseEnter, onMouseLeave
}) => {
  const effectiveOuterRadius = useMemo(() => {
    if (sector.isLeaf && sector.niveau) {
      const radiusDiff = outerRadius - innerRadius;
      const factor = niveaux[sector.niveau].radiusFactor;
      return innerRadius + (radiusDiff * factor);
    }
    return outerRadius;
  }, [sector, innerRadius, outerRadius]);

  const pathData = useMemo(() => {
    const startAngleRad = (startAngle * Math.PI) / 180;
    const endAngleRad = (endAngle * Math.PI) / 180;
    
    const x1 = centerX + innerRadius * Math.cos(startAngleRad);
    const y1 = centerY + innerRadius * Math.sin(startAngleRad);
    const x2 = centerX + effectiveOuterRadius * Math.cos(startAngleRad);
    const y2 = centerY + effectiveOuterRadius * Math.sin(startAngleRad);
    const x3 = centerX + effectiveOuterRadius * Math.cos(endAngleRad);
    const y3 = centerY + effectiveOuterRadius * Math.sin(endAngleRad);
    const x4 = centerX + innerRadius * Math.cos(endAngleRad);
    const y4 = centerY + innerRadius * Math.sin(endAngleRad);
    
    const largeArc = endAngle - startAngle > 180 ? 1 : 0;
    
    return `
      M ${x1} ${y1}
      L ${x2} ${y2}
      A ${effectiveOuterRadius} ${effectiveOuterRadius} 0 ${largeArc} 1 ${x3} ${y3}
      L ${x4} ${y4}
      A ${innerRadius} ${innerRadius} 0 ${largeArc} 0 ${x1} ${y1}
      Z
    `;
  }, [centerX, centerY, innerRadius, effectiveOuterRadius, startAngle, endAngle]);

  const fillOpacity = isActive 
    ? (sector.isLeaf && sector.niveau ? 0.5 + (sector.niveau * 0.125) : 0.85)
    : 0.2;

  return (
    <g>
      <path 
        d={pathData} 
        fill={color}
        fillOpacity={fillOpacity}
        stroke={isSelected || isHovered ? "white" : "#333"}
        strokeWidth={isSelected ? 3 : isHovered ? 2 : 0.5}
        style={{ 
          cursor: 'pointer',
          transition: 'all 0.2s ease'
        }}
        onClick={onClick}
        onMouseEnter={onMouseEnter}
        onMouseLeave={onMouseLeave}
      />
    </g>
  );
};

              <svg width={svgSize.width} height={svgSize.height} className="bg-white rounded-2xl shadow-2xl">
                <defs>
                  <radialGradient id="centerGradient">
                    <stop offset="0%" stopColor="#4c1d95" />
                    <stop offset="100%" stopColor="#7c3aed" />
                  </radialGradient>
                </defs>
                
                <g>
                  {currentView && currentView.children && renderSectorsRecursive(
                    currentView.children,
                    -90,
                    270,
                    0,
                    getMaxDepth(currentView),
                    null
                  )}
                </g>
                
                <circle
                  cx={centerX}
                  cy={centerY}
                  r={innerRadius}
                  fill="url(#centerGradient)"
                  stroke="white"
                  strokeWidth="3"
                  style={{ cursor: 'pointer' }}
                  onClick={handleReset}
                />
                
                <text 
                  x={centerX} 
                  y={centerY} 
                  textAnchor="middle" 
                  fontSize="12" 
                  fill="white"
                  style={{ pointerEvents: 'none' }}
                >
                  {Math.round(stats.points)}
                </text>
              </svg>
            </div>
          ) : (
            /* Vue temporelle */
            <div className="flex-1 overflow-auto p-4">
              <div className="bg-white rounded-xl p-4">
                <div ref={calendarRef}>
                  {/* Calendrier GitHub style */}
                  <div className="flex gap-[3px]">
                    <div className="flex flex-col gap-[3px] mr-2 text-[10px] text-gray-600">
                      {['', 'Lun', '', 'Mer', '', 'Ven', ''].map((day, i) => (
                        <div key={i} className="h-[11px] w-6 text-right pr-1">{day}</div>
                      ))}
                    </div>
                    
                    <div className="flex gap-[3px]">
                      {calendarGrid.map((week, weekIndex) => (
                        <div key={weekIndex} className="flex flex-col gap-[3px]">
                          {week.map((day, dayIndex) => {
                            if (!day.isInYear || day.isFuture) {
                              return <div key={dayIndex} className="w-[11px] h-[11px]" />;
                            }
                            
                            const bgColor = day.activity 
                              ? (day.activity.domains.size === 1 
                                  ? Array.from(day.activity.domains)[0] 
                                  : `rgb(${Math.max(100, 220 - day.activity.count * 20)}, ${Math.max(100, 220 - day.activity.count * 20)}, ${Math.max(100, 220 - day.activity.count * 20)})`)
                              : '#ebedf0';
                            
                            return (
                              <div
                                key={dayIndex}
                                className="w-[11px] h-[11px] rounded-sm cursor-pointer hover:outline hover:outline-1 hover:outline-gray-400"
                                style={{ 
                                  backgroundColor: bgColor,
                                  opacity: day.activity ? 0.6 + (day.activity.points / 100) : 1
                                }}
                                title={`${day.date.toLocaleDateString('fr-FR')}: ${
                                  day.activity ? `${day.activity.count} activité(s), ${day.activity.points} points` : 'Aucune activité'
                                }`}
                                onClick={() => handleDateClick(day.date)}
                              />
                            );
                          })}
                        </div>
                      ))}
                    </div>
                  </div>
                  
                  {/* Légende */}
                  <div className="mt-4 flex items-center gap-4 text-xs">
                    <span className="text-gray-600">Moins</span>
                    <div className="flex gap-1">
                      <div className="w-3 h-3 bg-gray-200 rounded-sm"></div>
                      <div className="w-3 h-3 bg-blue-300 rounded-sm"></div>
                      <div className="w-3 h-3 bg-blue-500 rounded-sm"></div>
                      <div className="w-3 h-3 bg-blue-700 rounded-sm"></div>
                      <div className="w-3 h-3 bg-blue-900 rounded-sm"></div>
                    </div>
                    <span className="text-gray-600">Plus</span>
                  </div>
                </div>
              </div>
            </div>
          )}

          {/* Statistiques */}
          <div className="p-4 bg-white/80 backdrop-blur border-t">
            <div className="grid grid-cols-5 gap-4">
              <div className="flex items-center gap-2">
                <TrendingUp className="text-green-500" size={20} />
                <div>
                  <p className="text-xs text-gray-600">Progression</p>
                  <p className="text-lg font-bold text-green-600">{competenceData.metadata.progressionTrimestre}</p>
                </div>
              </div>
              <div className="flex items-center gap-2">
                <Clock className="text-blue-500" size={20} />
                <div>
                  <p className="text-xs text-gray-600">Heures</p>
                  <p className="text-lg font-bold">{competenceData.metadata.heuresFormation}h</p>
                </div>
              </div>
              <div className="flex items-center gap-2">
                <BookOpen className="text-purple-500" size={20} />
                <div>
                  <p className="text-xs text-gray-600">Activités</p>
                  <p className="text-lg font-bold">{stats.visible}/{stats.total}</p>
                </div>
              </div>
              <div className="flex items-center gap-2">
                <Sparkles className="text-yellow-500" size={20} />
                <div>
                  <p className="text-xs text-gray-600">Points</p>
                  <p className="text-lg font-bold">{Math.round(stats.points)}</p>
                </div>
              </div>
              <div className="flex items-center gap-2">
                <Target className="text-red-500" size={20} />
                <div>
                  <p className="text-xs text-gray-600">Compétences</p>
                  <p className="text-lg font-bold">{competenceData.root.competenceCount}</p>
                </div>
              </div>
            </div>
          </div>
        </div>

        {/* Panel droit - Détails */}
        <div className="w-80 bg-white/90 backdrop-blur border-l flex flex-col">
          <div className="p-4 border-b bg-gradient-to-r from-blue-50 to-purple-50">
            {selectedNode ? (
              <div>
                <h2 className="text-lg font-bold mb-2 flex items-center gap-2">
                  {selectedNode.niveau > 0 && (
                    <span>{niveaux[selectedNode.niveau]?.icon}</span>
                  )}
                  {selectedNode.id}
                </h2>
                
                {selectedNode.description && (
                  <p className="text-sm text-gray-600 mb-3">{selectedNode.description}</p>
                )}
                
                <div className="space-y-2">
                  <div className="flex justify-between items-center p-2 bg-white rounded">
                    <span className="text-sm">Points</span>
                    <span className="font-bold text-blue-600">{Math.round(selectedNode.points)}</span>
                  </div>
                  
                  {selectedNode.niveau > 0 && (
                    <div className="flex justify-between items-center p-2 bg-white rounded">
                      <span className="text-sm">Niveau</span>
                      <div className="flex items-center gap-2">
                        <span className="font-bold" style={{ color: niveaux[selectedNode.niveau]?.color }}>
                          {niveaux[selectedNode.niveau]?.nom}
                        </span>
                        <div className="flex gap-0.5">
                          {[1, 2, 3, 4].map(level => (
                            <div
                              key={level}
                              className={`w-2 h-2 rounded-sm ${
                                selectedNode.niveau >= level ? 'bg-gradient-to-r from-blue-500 to-purple-500' : 'bg-gray-300'
                              }`}
                            />
                          ))}
                        </div>
                      </div>
                    </div>
                  )}

                  {selectedNode.isLeaf && selectedNode.traces && selectedNode.traces.length > 0 && (
                    <div className="p-2 bg-blue-50 rounded">
                      <span className="text-xs text-blue-700">
                        {selectedNode.traces.length} activité(s) réalisée(s)
                      </span>
                    </div>
                  )}

                  {selectedNode.competenceCount > 0 && (
                    <div className="flex justify-between items-center p-2 bg-white rounded">
                      <span className="text-sm">Sous-compétences</span>
                      <span className="font-bold">{selectedNode.competenceCount}</span>
                    </div>
                  )}
                </div>
              </div>
            ) : (
              <div className="text-center py-8">
                <Info className="mx-auto mb-3 text-gray-400" size={32} />
                <p className="text-gray-500">Sélectionnez un élément</p>
                <p className="text-xs text-gray-400 mt-2">
                  Cliquez sur un secteur pour voir les détails
                </p>
              </div>
            )}
          </div>

          {/* Liste des activités récentes */}
          <div className="flex-1 overflow-auto p-4">
            <h3 className="font-semibold mb-3 text-gray-700 flex items-center gap-2">
              <Activity size={16} />
              Activités récentes
              {selectedDate && (
                <button 
                  onClick={() => setSelectedDate(null)}
                  className="ml-auto text-xs text-blue-600 hover:text-blue-800"
                >
                  Réinitialiser
                </button>
              )}
            </h3>
            <div className="space-y-2">
              {xmlData
                .filter(trace => new Date(trace.date) <= (selectedDate ?? currentDate))
                .sort((a, b) => new Date(b.date) - new Date(a.date))
                .slice(0, 15)
                .map(trace => (
                  <div 
                    key={trace.id}
                    className="p-2 bg-gray-50 rounded hover:bg-gray-100 cursor-pointer text-sm transition-colors"
                    onClick={() => handleDateClick(new Date(trace.date))}
                  >
                    <div className="flex items-start gap-2">
                      <div 
                        className="w-3 h-3 rounded-full mt-0.5 flex-shrink-0"
                        style={{ backgroundColor: trace.domainColor }}
                      />
                      <div className="flex-1 min-w-0">
                        <div className="font-medium truncate">{trace.competence}</div>
                        <div className="text-xs text-gray-500 truncate">{trace.description}</div>
                        <div className="text-xs text-gray-400 mt-1">
                          {new Date(trace.date).toLocaleDateString('fr-FR')} • {trace.points} pts • {trace.type}
                        </div>
                      </div>
                    </div>
                  </div>
                ))}
            </div>
            {xmlData.filter(trace => new Date(trace.date) <= (selectedDate ?? currentDate)).length > 15 && (
              <p className="text-xs text-gray-500 text-center mt-3">
                Affichage des 15 dernières activités sur {xmlData.filter(trace => new Date(trace.date) <= (selectedDate ?? currentDate)).length}
              </p>
            )}
          </div>
        </div>
      </div>
    </div>
  );
};

export default UnifiedCompetenceViewer;
